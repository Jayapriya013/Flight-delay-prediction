def reprocess_flight_dataset(file_path):
    # Load the dataset
    df = pd.read_csv(file_path)
    
    # Step 1: Convert time columns to datetime

                     'Scheduled_Arrival_Time', 'Actual_Arrival_Time']
    for col in datetime_cols:
        df[col] = pd.to_datetime(df[col])
    
    # Step 2: Extract time-based features
    df['Dep_Hour'] = df['Scheduled_Departure_Time'].dt.hour
    df['Dep_DayOfWeek'] = df['Scheduled_Departure_Time'].dt.dayofweek

    df['Dep_Delay_Min'] = (df['Actual_Departure_Time'] - df['Scheduled_Departure_Time']).dt.total_seconds() / 60

    # Step 6: Drop original datetime and label columns
    df.drop(columns=datetime_cols + ['Flight_Status'], inplace=True)
    
    return df

import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
from tensorflow.keras.utils import to_categorical

# Synthetic data generation
# Rule: If the average of all wind speeds (feature 1) > 0.7, then delay (1), else on-time (0)
samples = 1200
timesteps = 6
features = 2

# Evaluate

y_true = np.argmax(y_test, axis=1)
